import type { EncontreiroQuadrante } from '@/@types/quadrante'
import React from 'react'

interface QuadrantePageProps {
  title: string
  description?: string
  members: EncontreiroQuadrante[]
  circleColor?: string
  isCoord?: boolean
  isPalestrante?: boolean
  children?: React.ReactNode
}

export function QuadrantePage({
  title,
  description,
  members,
  circleColor,
  isCoord = false,
  isPalestrante = false,
  children,
}: QuadrantePageProps) {
  const getTitleColor = () => {
    if (!circleColor) return 'text-black'
    return `text-circulo-${circleColor.toLowerCase()}`
  }

  return (
    <div className="sheet relative flex h-sheet w-sheet flex-wrap items-start justify-center p-10 print:mx-auto">
      <div className="flex w-full flex-wrap justify-center pt-2">
        <h1
          className={`font-custom w-full text-center text-4xl font-bold ${getTitleColor()}`}
        >
          {title}
        </h1>
        {description && (
          <p className="w-[600px] pb-5 pt-3 text-center text-sm italic">
            &quot;{description}&quot;
          </p>
        )}
      </div>

      {children || (
        <div className="flex flex-wrap items-center justify-center">
          {members.map((member, index) => (
            <div
              key={index}
              className={`w-[325px] p-2 text-sm ${isCoord ? 'font-bold' : ''}`}
            >
              <span className="font-bold">{member.nome}</span>
              <br />
              {!isPalestrante && (
                <>
                  <span>Apelido: {member.apelido}</span>
                  <br />
                  <span>
                    {circleColor ? 'Nascimento: ' : 'Encontr√£o: '}
                    {circleColor ? member.dataNasc : member.encontro}
                  </span>
                  <br />
                  <span>Telefone: {member.cel}</span>
                  <br />
                  <span>Bairro: {member.bairro}</span>
                  <br />
                  <span>Instagram: {member.insta}</span>
                </>
              )}
              {isPalestrante && <span>Tema: {member.tema}</span>}
            </div>
          ))}
        </div>
      )}
    </div>
  )
}
